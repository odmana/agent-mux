# Agent Mux Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a terminal multiplexer that manages multiple AI agent sessions with a tabbed sidebar, PTY-backed shell panes, and Claude Code notification integration.

**Architecture:** Three-layer stack: PTY layer (node-pty + xterm-headless per session), session manager (tab state, notifications, git branch), and renderer (raw ANSI escape sequences for sidebar + main pane). Input routing handles three contexts (navigation, terminal, directory picker) with a Ctrl+A prefix key and SGR mouse support.

**Tech Stack:** Node.js 22, TypeScript, pnpm, node-pty, @xterm/headless, @xterm/addon-serialize, vitest

### Task 2: PTY Manager

**Files:**
- Create: `src/pty-manager.ts`

The PTY manager wraps `node-pty` + `@xterm/headless` + `@xterm/addon-serialize` into a single unit per session. Each PTY session spawns a shell in a given directory, feeds output into a headless terminal for buffer management, and exposes a `serialize()` method to get the visible viewport as ANSI text.

No unit tests for this module — it's a thin wrapper around native PTY, so meaningful testing requires a real terminal. Verified manually in Task 6.

- [ ] **Step 1: Create `src/pty-manager.ts`**

```ts
import { spawn, type IPty } from 'node-pty';
import { Terminal } from '@xterm/headless';
import { SerializeAddon } from '@xterm/addon-serialize';

export interface PtySession {
  pty: IPty;
  terminal: Terminal;
  serializer: SerializeAddon;
}

export function createPtySession(
  cwd: string,
  cols: number,
  rows: number,
): PtySession {
  const shell = process.env.SHELL || '/bin/zsh';
  const pty = spawn(shell, [], {
    cwd,
    cols,
    rows,
    name: 'xterm-256color',
    env: { ...process.env, TERM: 'xterm-256color' },
  });

  const terminal = new Terminal({ cols, rows, allowProposedApi: true });
  const serializer = new SerializeAddon();
  terminal.loadAddon(serializer);

  pty.onData((data: string) => {
    terminal.write(data);
  });

  return { pty, terminal, serializer };
}

export function writeToPty(session: PtySession, data: string): void {
  session.pty.write(data);
}

export function resizePty(
  session: PtySession,
  cols: number,
  rows: number,
): void {
  session.pty.resize(cols, rows);
  session.terminal.resize(cols, rows);
}

export function serializeViewport(session: PtySession): string {
  return session.serializer.serialize({ scrollback: 0 });
}

export function killPty(session: PtySession): void {
  session.pty.kill();
  session.terminal.dispose();
}
```

Add to `src/renderer.ts`:

```ts
import type { NotificationState } from './session-manager.js';

const SIDEBAR_WIDTH = 30;
const DOT = '\u25CF'; // ●

export function renderSidebarRow(
  dirName: string,
  branch: string,
  notification: NotificationState,
  isActive: boolean,
  width: number = SIDEBAR_WIDTH,
): string {
  let out = '';

  // Notification dot: 2 chars (dot + space) or 2 spaces
  const showDot =
    notification === 'permission' ||
    (notification === 'idle' && !isActive);

  if (showDot) {
    const color = notification === 'permission' ? ansi.fg(1) : ansi.fg(4);
    out += `${color}${DOT}${ansi.reset()} `;
  } else {
    out += '  ';
  }

  // Truncate dir name and branch to fit within width
  // Layout: "DD DIRNAME     BRANCH" where DD is dot area (2 chars)
  const branchStr = branch ? ` ${branch}` : '';
  const availableForDir = width - 2 - branchStr.length;
  const displayDir =
    dirName.length > availableForDir
      ? dirName.slice(0, availableForDir - 1) + '\u2026'
      : dirName;
  const padding = ' '.repeat(
    Math.max(0, width - 2 - displayDir.length - branchStr.length),
  );

  if (isActive) {
    out = `${ansi.reverse()}${out}${displayDir}${padding}${ansi.dim()}${branchStr}${ansi.reset()}`;
  } else {
    out += `${displayDir}${padding}${ansi.dim()}${branchStr}${ansi.reset()}`;
  }

  return out;
}
```

Add to `src/renderer.ts`:

```ts
export interface TabInfo {
  dirName: string;
  branch: string;
  notification: NotificationState;
}

export class Renderer {
  private buffer: string = '';
  readonly sidebarWidth = SIDEBAR_WIDTH;

  private write(s: string): void {
    this.buffer += s;
  }

  private flush(): void {
    process.stdout.write(this.buffer);
    this.buffer = '';
  }

  get mainPaneCols(): number {
    return (process.stdout.columns || 80) - this.sidebarWidth - 1;
  }

  get rows(): number {
    return process.stdout.rows || 24;
  }

  enterAltScreen(): void {
    process.stdout.write(
      ansi.enterAltScreen() + ansi.hideCursor() + ansi.enableMouse() + ansi.clearScreen(),
    );
  }

  exitAltScreen(): void {
    process.stdout.write(
      ansi.disableMouse() + ansi.showCursor() + ansi.exitAltScreen(),
    );
  }

  render(
    tabs: TabInfo[],
    activeIndex: number,
    mainContent: string,
    inputMode: 'navigation' | 'terminal' | 'directory-picker',
  ): void {
    this.buffer = '';
    const totalRows = this.rows;
    const totalCols = process.stdout.columns || 80;
    const mainStart = this.sidebarWidth + 2; // 1-indexed, after divider

    // Draw sidebar rows
    for (let row = 0; row < totalRows; row++) {
      this.write(ansi.moveTo(row + 1, 1));
      this.write(ansi.clearLine());

      if (row < tabs.length) {
        const tab = tabs[row];
        this.write(
          renderSidebarRow(
            tab.dirName,
            tab.branch,
            tab.notification,
            row === activeIndex,
            this.sidebarWidth,
          ),
        );
      } else if (row === tabs.length) {
        // "+ New tab" row
        const label = '  + New tab';
        if (inputMode === 'navigation') {
          this.write(ansi.dim() + label + ansi.reset());
        } else {
          this.write(ansi.dim() + label + ansi.reset());
        }
      }

      // Divider
      this.write(ansi.moveTo(row + 1, this.sidebarWidth + 1));
      this.write(ansi.dim() + '\u2502' + ansi.reset()); // │
    }

    // Draw main pane
    const lines = mainContent.split('\n');
    for (let row = 0; row < totalRows; row++) {
      this.write(ansi.moveTo(row + 1, mainStart));
      if (row < lines.length) {
        // Write line, truncated to main pane width
        this.write(lines[row]);
      }
    }

    // Position cursor — only show in terminal mode
    if (inputMode === 'terminal' && tabs.length > 0) {
      this.write(ansi.showCursor());
    } else {
      this.write(ansi.hideCursor());
    }

    this.flush();
  }

  renderEmptyState(): void {
    this.buffer = '';
    const totalRows = this.rows;
    const mainStart = this.sidebarWidth + 2;

    for (let row = 0; row < totalRows; row++) {
      this.write(ansi.moveTo(row + 1, 1));
      this.write(ansi.clearLine());

      if (row === 0) {
        this.write(ansi.dim() + '  + New tab' + ansi.reset());
      }

      // Divider
      this.write(ansi.moveTo(row + 1, this.sidebarWidth + 1));
      this.write(ansi.dim() + '\u2502' + ansi.reset());
    }

    // Center message in main pane
    const msg = 'Press Ctrl+A, n to add a tab';
    const centerRow = Math.floor(totalRows / 2);
    const centerCol = mainStart + Math.floor((this.mainPaneCols - msg.length) / 2);
    this.write(ansi.moveTo(centerRow, centerCol));
    this.write(ansi.dim() + msg + ansi.reset());
    this.write(ansi.hideCursor());

    this.flush();
  }
}
```

```ts
import { randomUUID } from 'node:crypto';
import { execSync } from 'node:child_process';
import {
  createPtySession,
  killPty,
  type PtySession,
} from './pty-manager.js';

export type NotificationState = 'none' | 'idle' | 'permission';
export type InputMode = 'navigation' | 'terminal' | 'directory-picker';

export interface Session {
  id: string;
  ptySession: PtySession | null; // null in test mode
  directory: string;
  branch: string;
  notificationState: NotificationState;
}

interface SessionManagerOptions {
  createPty?: boolean; // false for testing without real PTYs
}

export class SessionManager {
  sessions: Session[] = [];
  activeIndex = -1;
  inputMode: InputMode = 'navigation';

  private readonly shouldCreatePty: boolean;

  constructor(options: SessionManagerOptions = {}) {
    this.shouldCreatePty = options.createPty !== false;
  }

  addSession(
    directory: string,
    cols?: number,
    rows?: number,
  ): Session {
    const ptySession = this.shouldCreatePty
      ? createPtySession(directory, cols ?? 80, rows ?? 24)
      : null;

    const session: Session = {
      id: randomUUID(),
      ptySession,
      directory,
      branch: this.getGitBranch(directory),
      notificationState: 'none',
    };

    this.sessions.push(session);
    this.activeIndex = this.sessions.length - 1;
    return session;
  }

  removeSession(id: string): void {
    const index = this.sessions.findIndex((s) => s.id === id);
    if (index === -1) return;

    const session = this.sessions[index];
    if (session.ptySession) {
      killPty(session.ptySession);
    }

    this.sessions.splice(index, 1);

    if (this.sessions.length === 0) {
      this.activeIndex = -1;
    } else if (this.activeIndex >= this.sessions.length) {
      this.activeIndex = this.sessions.length - 1;
    }
  }

  switchTo(index: number): void {
    if (index < 0 || index >= this.sessions.length) return;
    this.activeIndex = index;

    // Clear idle notifications on the tab we're switching to
    const session = this.sessions[index];
    if (session.notificationState === 'idle') {
      session.notificationState = 'none';
    }
  }

  getActive(): Session | null {
    if (this.activeIndex < 0 || this.activeIndex >= this.sessions.length) {
      return null;
    }
    return this.sessions[this.activeIndex];
  }

  setNotification(id: string, state: NotificationState): void {
    const session = this.sessions.find((s) => s.id === id);
    if (session) {
      session.notificationState = state;
    }
  }

  setNotificationByDirectory(
    directory: string,
    state: NotificationState,
  ): void {
    const session = this.sessions.find((s) => s.directory === directory);
    if (session) {
      // Skip idle notifications for the active tab
      if (
        state === 'idle' &&
        this.sessions.indexOf(session) === this.activeIndex
      ) {
        return;
      }
      session.notificationState = state;
    }
  }

  handlePtyOutput(id: string): void {
    const session = this.sessions.find((s) => s.id === id);
    if (session && session.notificationState !== 'none') {
      session.notificationState = 'none';
    }
  }

  getGitBranch(directory: string): string {
    try {
      return execSync('git rev-parse --abbrev-ref HEAD', {
        cwd: directory,
        encoding: 'utf-8',
        stdio: ['pipe', 'pipe', 'pipe'],
      }).trim();
    } catch {
      return '';
    }
  }
}
```

export interface MouseEvent {
  button: number;
  x: number;
  y: number;
  isRelease: boolean;
}

export function parseMouseEvent(buf: Buffer): MouseEvent | null {
  const str = buf.toString();
  const match = str.match(/^\x1b\[<(\d+);(\d+);(\d+)([Mm])$/);
  if (!match) return null;
  return {
    button: parseInt(match[1], 10),
    x: parseInt(match[2], 10),
    y: parseInt(match[3], 10),
    isRelease: match[4] === 'm',
  };
}

type SpecialKey =
  | 'ctrl-a'
  | 'ctrl-c'
  | 'ctrl-d'
  | 'up'
  | 'down'
  | 'enter'
  | 'escape'
  | 'tab'
  | 'backspace';

export function parseKey(buf: Buffer): SpecialKey | null {
  if (buf.length === 1) {
    switch (buf[0]) {
      case 0x01: return 'ctrl-a';
      case 0x03: return 'ctrl-c';
      case 0x04: return 'ctrl-d';
      case 0x09: return 'tab';
      case 0x0d: return 'enter';
      case 0x1b: return 'escape';
      case 0x7f: return 'backspace';
    }
    return null;
  }

  const str = buf.toString();
  if (str === '\x1b[A') return 'up';
  if (str === '\x1b[B') return 'down';
  return null;
}

export function setupRawMode(): void {
  if (process.stdin.isTTY) {
    process.stdin.setRawMode(true);
    process.stdin.resume();
  }
}

export function teardownRawMode(): void {
  if (process.stdin.isTTY) {
    process.stdin.setRawMode(false);
    process.stdin.pause();
  }
}

export interface InputActions {
  switchTab: (index: number) => void;
  focusTerminal: () => void;
  focusNavigation: () => void;
  writeToActivePty: (data: string) => void;
  openDirectoryPicker: () => void;
  removeActiveTab: () => void;
  quit: () => void;
  getInputMode: () => 'navigation' | 'terminal' | 'directory-picker';
  getActiveIndex: () => number;
  getTabCount: () => number;
  getSidebarWidth: () => number;
  onDirectoryPickerKey: (data: Buffer) => void;
}

export function createInputRouter(actions: InputActions): (data: Buffer) => void {
  let prefixPending = false;

  return (data: Buffer) => {
    const mode = actions.getInputMode();

    // Check for mouse events first (works in all modes)
    const mouse = parseMouseEvent(data);
    if (mouse && !mouse.isRelease) {
      if (mouse.x <= actions.getSidebarWidth()) {
        // Clicked sidebar
        if (mouse.y <= actions.getTabCount()) {
          actions.switchTab(mouse.y - 1);
          actions.focusTerminal();
        } else if (mouse.y === actions.getTabCount() + 1) {
          actions.openDirectoryPicker();
        }
      } else {
        // Clicked main pane
        if (mode === 'navigation') {
          actions.focusTerminal();
        } else if (mode === 'terminal') {
          // Forward mouse to PTY
          actions.writeToActivePty(data.toString());
        }
      }
      return;
    }

    if (mode === 'directory-picker') {
      actions.onDirectoryPickerKey(data);
      return;
    }

    // Handle prefix key
    if (prefixPending) {
      prefixPending = false;
      const key = parseKey(data);
      if (key === 'up' || key === 'down') {
        actions.focusNavigation();
        // Fall through to navigation handling below
      } else if (data.toString() === 'n') {
        actions.openDirectoryPicker();
        return;
      } else {
        // Unknown prefix combo — ignore
        return;
      }
    }

    const key = parseKey(data);

    if (mode === 'terminal') {
      if (key === 'ctrl-a') {
        prefixPending = true;
        return;
      }
      // Forward everything to PTY
      actions.writeToActivePty(data.toString());
      return;
    }

    // Navigation mode
    if (key === 'up') {
      const idx = actions.getActiveIndex();
      if (idx > 0) actions.switchTab(idx - 1);
    } else if (key === 'down') {
      const idx = actions.getActiveIndex();
      if (idx < actions.getTabCount() - 1) actions.switchTab(idx + 1);
    } else if (key === 'enter') {
      actions.focusTerminal();
    } else if (data.toString() === '+' || data.toString() === 'n') {
      actions.openDirectoryPicker();
    } else if (key === 'ctrl-c' || data.toString() === 'q') {
      actions.quit();
    }
  };
}

import { readdirSync, statSync } from 'node:fs';
import { resolve, dirname, basename } from 'node:path';
import { homedir } from 'node:os';

export class DirectoryPicker {
  input = '';
  suggestions: string[] = [];
  selectedIndex = 0;

  appendChar(ch: string): void {
    this.input += ch;
    this.selectedIndex = 0;
    this.updateSuggestions();
  }

  deleteChar(): void {
    if (this.input.length > 0) {
      this.input = this.input.slice(0, -1);
      this.selectedIndex = 0;
      this.updateSuggestions();
    }
  }

  acceptSuggestion(): void {
    if (this.suggestions.length > 0 && this.selectedIndex < this.suggestions.length) {
      this.input = this.suggestions[this.selectedIndex];
      this.selectedIndex = 0;
      this.updateSuggestions();
    }
  }

  confirm(): string {
    return this.input;
  }

  reset(): void {
    this.input = '';
    this.suggestions = [];
    this.selectedIndex = 0;
  }

  updateSuggestions(): void {
    if (this.input.length === 0) {
      this.suggestions = [];
      return;
    }

    const expanded = this.input.startsWith('~')
      ? homedir() + this.input.slice(1)
      : this.input;

    try {
      // If input ends with /, list contents of that directory
      if (expanded.endsWith('/')) {
        const entries = readdirSync(expanded, { withFileTypes: true });
        this.suggestions = entries
          .filter((e) => e.isDirectory() && !e.name.startsWith('.'))
          .map((e) => expanded + e.name)
          .slice(0, 8);
        return;
      }

      // Otherwise, list matching entries in the parent directory
      const parent = dirname(expanded);
      const prefix = basename(expanded).toLowerCase();
      const entries = readdirSync(parent, { withFileTypes: true });
      this.suggestions = entries
        .filter(
          (e) =>
            e.isDirectory() &&
            !e.name.startsWith('.') &&
            e.name.toLowerCase().startsWith(prefix),
        )
        .map((e) => resolve(parent, e.name))
        .slice(0, 8);
    } catch {
      this.suggestions = [];
    }
  }
}

import { readFileSync, writeFileSync, readdirSync, unlinkSync, existsSync, mkdirSync } from 'node:fs';
import { join } from 'node:path';
import { homedir } from 'node:os';
import type { SessionManager } from './session-manager.js';
import type { NotificationState } from './session-manager.js';

const SETTINGS_PATH = join(homedir(), '.claude', 'settings.json');
const MARKER = '# agent-mux';
const STATE_DIR = '/tmp';
const STATE_PREFIX = 'agent-mux-';
const STATE_SUFFIX = '.state';
const POLL_INTERVAL_MS = 500;

const HOOK_ENTRY = {
  matcher: 'idle_prompt|permission_prompt',
  hooks: [
    {
      type: 'command',
      command: `echo "$HOOK_EVENT_NAME $(pwd)" > "${STATE_DIR}/${STATE_PREFIX}$$.state" ${MARKER}`,
    },
  ],
};

export function ensureHookConfigured(): void {
  const claudeDir = join(homedir(), '.claude');
  if (!existsSync(claudeDir)) {
    mkdirSync(claudeDir, { recursive: true });
  }

  let settings: Record<string, unknown> = {};
  try {
    settings = JSON.parse(readFileSync(SETTINGS_PATH, 'utf-8'));
  } catch {
    // File doesn't exist or is invalid — start fresh
  }

  const hooks = (settings.hooks ?? {}) as Record<string, unknown[]>;
  const notifications = (hooks.Notification ?? []) as Array<{
    hooks?: Array<{ command?: string }>;
  }>;

  // Check if our hook already exists
  const alreadyConfigured = notifications.some((entry) =>
    entry.hooks?.some((h) => h.command?.includes(MARKER)),
  );

  if (alreadyConfigured) return;

  notifications.push(HOOK_ENTRY);
  hooks.Notification = notifications;
  settings.hooks = hooks;

  writeFileSync(SETTINGS_PATH, JSON.stringify(settings, null, 2) + '\n');
}

export function startNotificationWatcher(
  sessionManager: SessionManager,
  onUpdate: () => void,
): NodeJS.Timeout {
  return setInterval(() => {
    try {
      const files = readdirSync(STATE_DIR).filter(
        (f) => f.startsWith(STATE_PREFIX) && f.endsWith(STATE_SUFFIX),
      );

      for (const file of files) {
        const fullPath = join(STATE_DIR, file);
        try {
          const content = readFileSync(fullPath, 'utf-8').trim();
          const firstSpace = content.indexOf(' ');
          if (firstSpace === -1) continue;

          const eventName = content.slice(0, firstSpace);
          const directory = content.slice(firstSpace + 1);

          let state: NotificationState = 'none';
          if (eventName === 'permission_prompt') {
            state = 'permission';
          } else if (eventName === 'idle_prompt') {
            state = 'idle';
          }

          if (state !== 'none') {
            sessionManager.setNotificationByDirectory(directory, state);
            onUpdate();
          }

          // Delete the state file after processing
          unlinkSync(fullPath);
        } catch {
          // File may have been deleted between readdir and read
        }
      }
    } catch {
      // /tmp read failed — skip this poll cycle
    }
  }, POLL_INTERVAL_MS);
}

export function cleanupStateFiles(): void {
  try {
    const files = readdirSync(STATE_DIR).filter(
      (f) => f.startsWith(STATE_PREFIX) && f.endsWith(STATE_SUFFIX),
    );
    for (const file of files) {
      try {
